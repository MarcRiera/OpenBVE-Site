[Compressor]

[Suspension]

Right = FileName

[Horn]

Music = FileName

[Door]

Open Right = FileName

Close Right = FileName

[Ats]

[Buzzer]

[Pilot Lamp]

Down = FileName
Min = FileName
Max = FileName

[Reverser]
On = FileName
Off = FileName

[Breaker]
On = FileName
Off = FileName

[Others]
Noise = FileName
