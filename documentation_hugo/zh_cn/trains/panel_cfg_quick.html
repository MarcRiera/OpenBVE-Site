The panel.cfg file format - Quick reference

[Panel]
Background = FileName

[View]
Yaw = ValueInDegrees
Pitch = ValueInDegrees

[PressureIndicator]
Type = { 0 = gauge | 1 = LED }
LowerNeedle = Subject, RedValue, GreenValue, BlueValue
UpperNeedle = Subject, RedValue, GreenValue, BlueValue
Center = X, Y
Radius = ValueInPixels
Background = FileName
Cover = FileName
Unit = { 0 = kpa | 1 = kgf/cm2 }
Minimum = PressureValue
Maximum = PressureValue
Angle = ValueInDegrees

[SpeedIndicator]
Type = { 0 = gauge | 1 = LED }
Needle = RedValue, GreenValue, BlueValue
Center = X, Y
Radius = ValueInPixels
Background = FileName
Cover = FileName
Minimum = SpeedValue
Maximum = SpeedValue
Atc = FileName
AtcRadius = ValueInPixels

[DigitalIndicator]
Number = FileName
Corner = Left, Top
Size = Width, Height
Unit = { 0 = km/h | 1 = mph | 2 = m/s }

[PilotLamp]
TurnOn = FileName
TurnOff = FileName
Corner = Left, Top

[Watch]
Background = FileName
Center = X, Y
Radius = ValueInPixels
Needle = RedValue, GreenValue, BlueValue

[BrakeIndicator]
Image = FileName
Corner = Left, Top
Width = ValueInPixels
